      // Auto-reload for CDN propagation delays
      // Progressive retry: 100ms → 300ms → 500ms → 800ms → 1200ms → 2000ms (total ~2s)
      // Prevents slow connection loops with exponential backoff and max retries
      (function() {
        var RETRY_KEY = 'easyapp_retry';
        var RETRY_TIME_KEY = 'easyapp_retry_time';
        var MAX_RETRIES = 3;
        var reloading = false;
        var checkScheduled = false;
        // Progressive check intervals (cumulative: 100, 300, 600, 1000, 1500ms)
        // If root has content at any point, we stop checking
        var CHECK_INTERVALS = [100, 200, 300, 400, 500];
        var FINAL_CHECK = 2000; // Final check at 2s total (after all progressives)
        // Get retry count, reset if older than 2 minutes
        var lastRetry = sessionStorage.getItem(RETRY_TIME_KEY);
        if (lastRetry && Date.now() - parseInt(lastRetry) > 120000) {
          sessionStorage.removeItem(RETRY_KEY);
          sessionStorage.removeItem(RETRY_TIME_KEY);
        }
        var retries = parseInt(sessionStorage.getItem(RETRY_KEY) || '0');
        // Show user-friendly error when max retries exceeded (using safe DOM methods)
        function showLoadError() {
          var root = document.getElementById('root');
          if (!root) return;
          // Clear existing content safely
          while (root.firstChild) root.removeChild(root.firstChild);
          // Build error UI with DOM API (XSS-safe)
          var container = document.createElement('div');
          container.style.cssText = 'display:flex;flex-direction:column;align-items:center;justify-content:center;width:100%;height:100%;padding:20px;text-align:center;font-family:-apple-system,BlinkMacSystemFont,sans-serif;box-sizing:border-box;';
          var icon = document.createElement('div');
          icon.style.cssText = 'font-size:48px;margin-bottom:16px;';
          icon.textContent = '⚠️';
          var title = document.createElement('h2');
          title.style.cssText = 'margin:0 0 8px;font-size:18px;font-weight:600;';
          title.textContent = 'Page failed to load';
          var message = document.createElement('p');
          message.style.cssText = 'margin:0 0 20px;font-size:14px;color:#666;';
          message.textContent = 'Please reload the page or check your internet connection.';
          var button = document.createElement('button');
          button.style.cssText = 'padding:12px 24px;font-size:14px;font-weight:500;background:#007AFF;color:#fff;border:none;border-radius:8px;cursor:pointer;';
          button.textContent = 'Reload';
          button.onclick = function() {
            sessionStorage.clear();
            location.reload();
          };
          container.appendChild(icon);
          container.appendChild(title);
          container.appendChild(message);
          container.appendChild(button);
          root.appendChild(container);
        }
        function doReload() {
          if (reloading) return;
          // Show error UI if max retries exceeded
          if (retries >= MAX_RETRIES) {
            showLoadError();
            return;
          }
          reloading = true;
          sessionStorage.setItem(RETRY_KEY, String(retries + 1));
          sessionStorage.setItem(RETRY_TIME_KEY, String(Date.now()));
          // Nuclear option: Clear ALL caches and unregister service worker
          // This ensures fresh assets on next load
          var cleanupPromises = [];
          // 1. Delete all caches
          if (window.caches) {
            cleanupPromises.push(
              caches.keys().then(function(names) {
                return Promise.all(names.map(function(name) {
                  return caches.delete(name);
                }));
              })
            );
          }
          // 2. Unregister service worker (forces fresh SW on reload)
          if ('serviceWorker' in navigator) {
            cleanupPromises.push(
              navigator.serviceWorker.getRegistrations().then(function(registrations) {
                return Promise.all(registrations.map(function(reg) {
                  return reg.unregister();
                }));
              })
            );
          }
          // 3. Reload after cleanup (with error logging for debugging)
          Promise.all(cleanupPromises)
            .catch(function(err) {
              // Log cleanup errors for debugging (won't block reload)
              if (window.console && console.warn) {
                console.warn('[EasyApp] Cache cleanup error:', err);
              }
            })
            .finally(function() {
              window.location.reload();
            });
        }
        function isRootPopulated() {
          var root = document.getElementById('root');
          return root && root.children.length > 0;
        }
        // Method 1: Catch script/module loading errors immediately (0ms detection)
        // Handles both classic scripts and ES modules
        window.addEventListener('error', function(e) {
          // Classic script tag errors (target is the script element)
          if (e.target && e.target.tagName === 'SCRIPT' && e.target.src) {
            doReload();
            return;
          }
          // ES Module errors - target may be window, check for src property
          // This catches cases where Firefox reports module load failures
          // Only reload for JavaScript files - other resource errors (images, API, etc.) should not trigger reload
          if (e.target && e.target.src && typeof e.target.src === 'string' &&
            (e.target.src.indexOf('.js') !== -1 || e.target.tagName === 'SCRIPT')) {
            doReload();
            return;
          }
          // Browser-specific error messages for module/script loading failures
          // Firefox: "Loading failed for the module with source..."
          // Chrome: "Failed to load module script...", "ChunkLoadError"
          // Safari: "Could not load module script", "Script error.", "status of 404"
          var errorText = e.message || (e.error && e.error.message) || String(e);
          if (errorText && (
              // Firefox
              errorText.indexOf('Loading failed') !== -1 ||
              // Chrome
              errorText.indexOf('Failed to load module') !== -1 ||
              errorText.indexOf('Failed to fetch') !== -1 ||
              errorText.indexOf('ChunkLoadError') !== -1 ||
              // Safari
              errorText.indexOf('Could not load') !== -1 ||
              errorText.indexOf('Script error') !== -1 ||
              errorText.indexOf('status of 404') !== -1 ||
              // Common
              errorText.indexOf('MIME type') !== -1 ||
              errorText.indexOf('Unexpected token') !== -1
            )) {
            doReload();
            return;
          }
          // Firefox filename check for module errors
          if (e.filename && e.filename.indexOf('.js') !== -1 && e.lineno === 0) {
            doReload();
          }
        }, true);
        // Method 1b: Catch dynamic import failures (Promise rejections)
        // Handles React.lazy() and other dynamic imports
        window.addEventListener('unhandledrejection', function(e) {
          var reason = e.reason;
          if (reason && (
              (reason.name && reason.name === 'ChunkLoadError') ||
              (reason.message && (
                reason.message.indexOf('Loading chunk') !== -1 ||
                reason.message.indexOf('Failed to fetch dynamically imported module') !== -1 ||
                reason.message.indexOf('Failed to load module') !== -1 ||
                reason.message.indexOf('Loading failed') !== -1
              ))
            )) {
            e.preventDefault(); // Prevent console error
            doReload();
          }
        });
        // Method 2: Progressive root checks with early exit
        function scheduleChecks() {
          if (checkScheduled) return;
          checkScheduled = true;
          var currentDelay = 0;
          // Schedule progressive checks
          CHECK_INTERVALS.forEach(function(interval) {
            currentDelay += interval;
            setTimeout(function() {
              // Early exit if root is populated
              if (isRootPopulated()) {
                clearSuccessState();
                return;
              }
            }, currentDelay);
          });
          // Final check - trigger reload if still empty
          setTimeout(function() {
            if (!isRootPopulated()) {
              doReload();
            }
          }, FINAL_CHECK);
        }
        function clearSuccessState() {
          sessionStorage.removeItem(RETRY_KEY);
          sessionStorage.removeItem(RETRY_TIME_KEY);
        }
        // Start progressive checks when DOM is ready
        if (document.readyState === 'loading') {
          document.addEventListener('DOMContentLoaded', scheduleChecks);
        } else {
          scheduleChecks();
        }
        // Clear retry count on successful load
        window.addEventListener('load', function() {
          setTimeout(function() {
            if (isRootPopulated()) {
              clearSuccessState();
            }
          }, 500);
        });
      })();
